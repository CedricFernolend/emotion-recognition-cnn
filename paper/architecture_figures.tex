% Define styles
\tikzset{
    block/.style={
        rectangle,
        draw=black,
        thick,
        fill=white,
        minimum width=1.8cm,

        font=\small\bfseries

        rectangle,
        draw=black,
        thick,
        fill=gray!20,
        minimum width=1cm,
        minimum height=0.7cm,
        align=center,
        font=\scriptsize\bfseries

        rectangle,
        draw=black,
        thick,
        fill=gray!40,
        minimum width=0.8cm,
        minimum height=0.6cm,
        align=center,
        font=\scriptsize
    },

    },
    arrow/.style={
        ->,
        >=Stealth,
        thick
    },
    skiparrow/.style={
        ->,
        >=Stealth,
        thick,
        dashed
    },
    label/.style={
        font=\scriptsize,
        text=gray
    }
}
